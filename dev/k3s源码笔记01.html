# build流程

  1. download 下载所需要的包
    1. version.sh 获取所需包的版本号
    2. git cne --single-branch --branch=v1.1.8 --depth=1 https://github.com/opencontainers/runc build/src/github.com/opencontainers/runc
    3. https://github.com/k3s-io/k3s-root/releases/download/v0.12.2/k3s-root-amd64.tar
    4. git clone --single-branch --branch=v1.7.7-k3s1 --depth=1 https://github.com/k3s-io/containerd build/src/github.com/containerd/containerd
    5. curl -sfL https://k3s.io/k3s-charts/assets/traefik-crd/traefik-crd-25.0.2+up25.0.0.tgz -o build/static/charts/traefik-crd-25.0.2+up25.0.0.tgz
    6. curl -sfL https://k3s.io/k3s-charts/assets/traefik/traefik-25.0.2+up25.0.0.tgz -o build/static/charts/traefik-25.0.2+up25.0.0.tgz
  2. generate
    1. rm -rf build/data && mkdir -p build/data
    2. go generate
  3. SKIP_VALIDATE=true make
    1. make
      1. .dapper 下载.dapper这个工具
      2. .dapper ci
        1. ./download 与上面的第一步重复
        2. ./validate # SKIP_VALIDATE=true 跳过validate部分，主要就是不执行golangci-lint run
        3. ./build
          1. 删除可能存在的旧文件
          2. build cni
            1. go build -tags 'static_build libsqlite3 ctrd apparmor seccomp netcgo osusergo providerless urfave_cli_no_docs selinux' -gcflags=all= -ldflags '-X ... -w -s -extldflags '\''-static -lm -ldl -lz -lpthread'\'' ' -o /go/src/github.com/k3s-io/k3s/bin/cni
          3. build k3s
            1. go build -tags 'static_build libsqlite3 ctrd apparmor seccomp netcgo osusergo providerless urfave_cli_no_docs selinux' -buildvcs=false -gcflags=all= -ldflags '-X...... -w -s -extldflags '\''-static -lm -ldl -lz -lpthread'\'' ' -o bin/k3s ./cmd/server
          4. build containerd-shim bin/containerd-shim-runc-v2
          5. build runc ./bin/runc
        4. ./package
          1. 检查 ../bin/containerd 是否存在，不存在就执行./build
          2. package-ci
            1. 生成.tar.zst
            2. CGO_ENABLED=0 go build -tags urfave_cli_no_docs -buildvcs=false -ldflags ' -X github.com/k3s-io/k3s/pkg/version.Version=v1.28.4+k3s-3f237230 -X github.com/k3s-io/k3s/pkg/version.GitCommit=3f237230 -w -s -extldflags '\''-static'\''' -o dist/artifacts/k3s ./cmd/k3s
            3. 上传到aws s3 检查到没有配置key,会跳过
          3. package-image
            1. docker save docker.io/rancher/klipper-helm:v0.8.2-build20230815 docker.io/rancher/klipper-lb:v0.4.4 docker.io/rancher/local-path-provisioner:v0.0.24 docker.io/rancher/mirrored-coredns-coredns:1.10.1 docker.io/rancher/mirrored-library-busybox:1.36.1 docker.io/rancher/mirrored-library-traefik:2.10.5 docker.io/rancher/mirrored-metrics-server:v0.6.3 docker.io/rancher/mirrored-pause:3.6 -o dist/artifacts/k3s-airgap-images-amd64.tar
            2. zstd压缩
            3. trivy扫描
          4. package-airgap
            1. docker pull
            2. docker save .tar
            3. zstd .tar -> .tar.zst
            4. pigz .tar -> .tar.gz
        5. binary_size_check.sh
          1. 检查k3s文件大小是不是超过70m

TARGETS := $(shell ls scripts | grep -v \\.sh)  # scripts下的全部非sh文件
GO_FILES ?= $$(find . -name '*.go' | grep -v generated)   # 当前目录下的所有go文件（排除generated后）

.dapper:  # 从github上下载dapper
	@echo Downloading dapper
	@curl -sL https://releases.rancher.com/dapper/v0.6.0/dapper-$$(uname -s)-$$(uname -m) > .dapper.tmp
	@@chmod +x .dapper.tmp
	@./.dapper.tmp -v
	@mv .dapper.tmp .dapper

$(TARGETS): .dapper
	./.dapper $@  

.PHONY: deps
deps:
	go mod tidy

release:
	./scripts/release.sh

.DEFAULT_GOAL := ci  # 默认执行的target
