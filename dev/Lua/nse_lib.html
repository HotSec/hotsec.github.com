# 用C／C＋＋编写NSE模块

一个简单的NSE　Ｃ模块的步骤：

  1. 将头文件和源文件放在nmap根目录下

  2. 向makefile.in文件中添加源文件、头文件和对象文件

  3. 在NSE.main.cc文件中对新库文件进行连接操作

新建一个库test，创建nse_test.h、nse_test.cc.

// nse_test.cc
extern "C" {
    #include "lauxlib.h"
    #include "lua.h"
}

#include "nse_test.h"

static int hello_world(lua_State *L) {
    printf("Hello World From a C library\n");
    return 1;
}

static const struct luaL_Reg testlib[] = {
    { "hello", hello_world}
    {NULL,NULL}
};

LUALIB_API int luaopen_test(lua_State *L){
    luaL_newlib(L,testlib);
    return 1;
}



// nse_test.h
#ifndef TESTLIB
#define TESTLIB
#define TEsTLIBNAME "test"

LUALIB_API int luaopen_test(lua_State *L);

#endif

# 常见的NSE库文件

# shortport

这个库的目的是帮助建立port规则，它会收集常见端口规则信息。

shortport中提供了port_or_service(port,services,protos,states)方法。 
    - ports: 一个或多个端口号
    - services: 一个或多个服务的名称
    - protos: 一个或多个匹配的协议，默认值为“tcp"
    - states: 一个或者多个匹配的状态，默认值为{"open","open|filtered"}

这个方法在接到扫描结果之后，如果检测到参数中提供的ports、services，返回值就为true.

exchange(host,port,data,opts)
    - host: 连接的目标主机
    - port: 目标主机的指定端口
    - data: 开始要发送的数据
    - opts: 指定传递中的一些细节，可选项   bytes: 读取的最小字节数;  lines: 读取的最小行数
    - 函数的返回值有两个  status(true/false) ; 数据或者error

local shortport = require "shortport"

local comm = require "comm"
description = [[shortport使用的示例代码]]

author = "V5"
license = "Same as Nmap--See http://nmap.org/book/man-legal.html"
categories = {"default"}


portrule = shortport.port_or_service(13,"daytime",{"tcp","udp"})


action = function(host, port)

 local status, result = comm.exchange(host,port,"dummy",{lines=1})
 if status then
  return result
 end
end

# http

local shortport = require "shortport"

local http = require "http"
local stdnse = require "stdnse"
description = [[http的示例
]]

author = "V5"
license = "Same as Nmap--See http://nmap.org/book/man-legal.html"
categories = {"default"}


portrule = shortport.http


action = function(host, port)
 local uri = "/anony/mjpg.cgi"
 local _,status_404,resp_404 = http.identify_404(host,port)
 if status_404 == 200 then
  stdnse.print_debug(1, 
  "%s: Web server returns ambiguous response. Trendnet webcams return standard 404 status responses. Exiting.",
  SCRIPT_NAME)
  return 
 end
 stdnse.print_debug(1,"%s: HTTP HEAD %s", SCRIPT_NAME,uri)
 local resp = http.head(host,port,uri)
 if resp.status and resp.status == 200 then
  return string.format(
   "Trendnet IV-IP 110W video feed is unprotected:http://%s/anony/mjpg.cgi",
   host.ip)
 end
end

# openssl

用来绑定到openssl的接口，主要用来进行加密，它是否可用取决于nmap的建立方式，但是可以通过pcall()的帮助来查看这个类是否可用。

if not pcall(require,"openssl") then 
    action = function(host,port)
        stdnse.print_debug(2,"Skipping \"%s\" because OpenSSL is missing.",id)
    end
end

action = action or function(host,port)
    ...
end

# target

管理新发现目标的扫描队列。NSE根据prerule、hostrule、portrult规则来向当前的扫描队列中添加新的目标。

local status,err = target.add("192.168.1.1","192.168.1.2",...)

# creds

用来保存脚本在执行时发现的登录凭证。这些凭证都被保存在注册表中，同样它也提供了一个和数据库协同合作的接口。

local c = creds.credentials:new(SCRIPT_NAME,host,port)
c:add("packtpub","secret",creds.State.VALLID)
